class  LinkNode
{
	Data someData;
	LinkNode * nextNode;
}

bool isCircular(LinkNode * i_pRoot)
{
	if(i_pRoot)
		return null;

	LinkNode* fast = i_pRoot;
	LinkNode * slow = i_pRoot;
	bool slowGo = false;
	while(fast!=slow && fast->nextNode)
	{
		if(slowGo)
		{
			slow = slow->nextNode;
		}
		fast = fast->nextNode;

		slowGo = !slowGo;
	}

	return(slow == fast);
}
